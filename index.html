const colors = ["red", "blue", "green", "yellow", "purple", "orange"]; let targetColor = ""; let score = 0; document.addEventListener("DOMContentLoaded", () => { startNewGame(); }); function startNewGame() { targetColor = colors[Math.floor(Math.random() * colors.length)]; document.getElementById("colorBox").style.backgroundColor = targetColor; document.getElementById("score").textContent = score; document.getElementById("gameStatus").textContent = ""; generateColorOptions(); } function generateColorOptions() { const optionsContainer = document.getElementById("colorOptions"); optionsContainer.innerHTML = ""; colors.forEach(color => { const button = document.createElement("button"); button.style.backgroundColor = color; button.setAttribute("data-testid", "colorOption"); button.textContent = color; button.addEventListener("click", () => checkGuess(color)); optionsContainer.appendChild(button); }); } function checkGuess(selectedColor) { if (selectedColor === targetColor) { score++; document.getElementById("gameStatus").textContent = "Correct!"; setTimeout(startNewGame, 1000); // Reset color after win } else { document.getElementById("gameStatus").textContent = "Wrong! Try again."; } document.getElementById("score").textContent = score; } document.getElementById("newGameButton").addEventListener("click", startNewGame);
